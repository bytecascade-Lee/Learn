8        public static void main(String[] args) {

10           定义一个长度为3的数组，数组存储1~3名学生对象作为初始数据，学生对象的学号，姓名各不相同。 
11           学生的属性:学号，姓名，年龄。 
12           要求1:再次添加一个学生对象，并在添加的时候进行学号的唯一性判断。 
13           要求2:添加完毕之后，遍历所有学生信息。 
14           要求3:通过id删除学生信息，如果存在，则删除，如果不存在，则提示删除失败。 
15           要求4:删除完毕之后，遍历所有学生信息。 
16           要求5:查询数组id为“2”的学生，如果存在，则将他的年龄 +1岁 
17           */
18   
19   
20           //1.创建一个长度为3的数组
21           Student[] arr = new Student[3];
22   
23           //2.创建学生对象
24           Student stu1 = new Student(1, "小明", 18);
25           Student stu2 = new Student(2, "小红", 19);
26           Student stu3 = new Student(3, "小兰", 20);
27   
28           //3.把学生对象添加到数组当中
29           arr[0] = stu1;
30           arr[1] = stu2;
31           arr[2] = stu3;
32   
33           //4.再次创建一个学生对象
34           Student stu4 = new Student(4, "小智", 17);
35   
36           //5.唯一性判断
37           boolean flag = checkOnly(arr, stu4.getId());
38           if (flag) {
39               //5.2不存在---添加学生对象
40               //6.添加学生对象
41               int count = getCount(arr);
42               if (count == arr.length) {
43                   //6.1 老数组已经存满
44                   arr = createNewArray(arr, 1);
45                   int index = getFirstNullIndex(arr);
46                   arr[index] = stu4;
47   
48               } else {
49                   //6.2老数组没有存满
50                   int index = getFirstNullIndex(arr);
51                   arr[index] = stu4;
52               }
53   
54           } else {
55               //5.1已存在---提示重复
56               System.out.println("已存在，添加失败。");
57           }
58   
59   
60           //7通过id删除学生信息，如果存在，则删除，如果不存在，则提示删除失败。
61           //7.1存在---删除
62           //7.2不存在---提示失败
63           deleteStudent(arr, 1);
64   
65           //删除完毕之后，遍历所有学生信息。
66           printStudents(arr);
67   
68           //查询数组id为“2”的学生，如果存在，则将他的年龄 +1岁
69           for (Student stu : arr) {
70               if (stu != null) {
71                   if (stu.getId() == 2) {
72                       stu.setAge(stu.getAge() + 1);
73                   }
74               }
75           }

80       public static void deleteStudent(Student[] arr, int id) {
81           for (int i = 0; i < arr.length; i++) {
82               Student stu = arr[i];
83               if (stu != null) {
84                   if (stu.getId() == id) {
85                       arr[i] = null;
86                       return;
87                   }
88               }
89               System.out.println("id不存在，删除失败");
90           }
91   
92       }
93   
94       public static void printStudents(Student[] arr) {
95           for (Student stu : arr) {
96               if (stu != null) {
97                   System.out.print("学号：");
98                   System.out.print(stu.getId());
99                   System.out.print(", 姓名：");
100                  System.out.print(stu.getName());
101                  System.out.print(", 年龄：");
102                  System.out.println(stu.getAge());
103              }
104          }
105  
106      }

108      public static Student[] createNewArray(Student[] arr, int plusLength) {
109          Student[] newArr = new Student[arr.length + plusLength];
110          for (int i = 0; i < arr.length; i++) {
111              newArr[i] = arr[i];
112          }
113          return newArr;
114      }
115  
116      public static int getFirstNullIndex(Student[] arr) {
117          for (int i = 0; i < arr.length; i++) {
118              if (arr[i] == null) {
119                  return i;
120              }
121          }
122          return -1;
123  
124      }

126      public static boolean checkOnly(Student[] arr, int id) {
127          for (Student stu : arr) {
128              if (stu != null) {
129                  if (stu.getId() == id) {
130                      return false;
131                  }
132              }
133          }
134          return true;
135  
136      }

138      public static int getCount(Student[] arr) {
139          int count = 0;
140          for (Student stu : arr) {
141              if (stu != null) {
142                  count++;
143              }
144          }
145          return count;
